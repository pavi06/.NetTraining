def merge_the_tools(string, k):
    # your code goes here
    for i in range(0, len(string), k):
        resString=""
        for i in string[i:i+k]:
            if i not in resString:
                resString+=i
        print(resString)

if __name__ == '__main__':
    string, k = input(), int(input())
    merge_the_tools(string, k)
