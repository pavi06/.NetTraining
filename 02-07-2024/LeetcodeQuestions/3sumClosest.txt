class Solution:
    def threeSumClosest(self, nums: List[int], target: int) -> int:
        nums.sort()
        target_gap = float('inf')
        for i in range(len(nums)-2): 
            l,r = i+1, len(nums)-1
            new_target = target - nums[i]
            while l<r:
                sum_of_other_two = nums[l]+nums[r]
                if abs(target_gap) > abs(new_target - sum_of_other_two):
                    target_gap = new_target - sum_of_other_two
                
                if sum_of_other_two == new_target:
                    return target
                elif sum_of_other_two < new_target:
                    l+=1
                else:
                    r-=1
        return target - target_gap
